О задаче

Автоматическая классификация 7 типов стекла (оконное, автомобильное, контейнерное и т.д.) по 9 химическим параметрам. Применимо в системах контроля качества на стекольном производстве. Данные: UCI Glass Identification Dataset — 214 образцов, 9 признаков (Na, Mg, Al, Si, K, Ca, Ba, Fe + показатель преломления). Подход: - 5 моделей с 10-fold cross-validation (GLM, CART, Random Forest, SVM, Neural Network) - Grid search для подбора гиперпараметров - Бизнес-оценка через Profit Curve с учётом стоимости ошибок

4.2. Функции построения кривых

plot_lift_gain <- function(data, prob_col, actual_col, class_label) {
  data <- data %>%
    mutate(actual_binary = ifelse({{actual_col}} == class_label, 1, 0),
           prob = {{prob_col}}) %>%
    arrange(desc(prob)) %>%
    mutate(
      row_id = row_number(),
      total_n = n(),
      percentile = row_id / total_n,
      cum_found = cumsum(actual_binary),
      total_found = sum(actual_binary),
      gain = cum_found / total_found,
      lift = gain / percentile
    )
  
  p_gain <- ggplot(data, aes(x = percentile * 100, y = gain * 100)) +
    geom_line(color = "steelblue", size = 1) +
    geom_line(data = tibble(percentile = c(0, 100), gain = c(0, 100)),
              aes(x = percentile, y = gain), linetype = "dashed", color = "gray") +
    labs(title = paste("Gain Curve:", class_label),
         x = "% выборки", y = "% найденных целевых объектов") +
    theme_minimal()
  
  p_lift <- ggplot(data %>% filter(percentile > 0), 
                   aes(x = percentile * 100, y = lift)) +
    geom_line(color = "darkgreen", size = 1) +
    geom_hline(yintercept = 1, linetype = "dashed", color = "gray") +
    labs(title = paste("Lift Curve:", class_label),
         x = "% выборки", y = "Lift (во сколько раз лучше случайного)") +
    theme_minimal() +
    ylim(0, NA)
  
  return(list(gain = p_gain, lift = p_lift))
}

plot_profit_curve <- function(data, prob_col, actual_col, class_label,
                              profit_tp = 10, cost_fp = -2, cost_fn = -5) {
  profit_data <- data %>%
    mutate(actual_binary = ifelse({{actual_col}} == class_label, 1, 0),
           prob = {{prob_col}}) %>%
    arrange(desc(prob))
  
  thresholds <- seq(0, 1, length.out = 100)
  profit_results <- map_dfr(thresholds, function(thr) {
    preds <- ifelse(profit_data$prob >= thr, 1, 0)
    tp <- sum(preds == 1 & profit_data$actual_binary == 1)
    fp <- sum(preds == 1 & profit_data$actual_binary == 0)
    fn <- sum(preds == 0 & profit_data$actual_binary == 1)
    total_profit <- (tp * profit_tp) + (fp * cost_fp) + (fn * cost_fn)
    percentile_selected <- mean(profit_data$prob >= thr)
    
    tibble(
      threshold = thr,
      percentile = percentile_selected * 100,
      profit = total_profit,
      tp = tp, fp = fp, fn = fn
    )
  })
  
  p_profit <- ggplot(profit_results, aes(x = percentile, y = profit)) +
    geom_line(color = "purple", size = 1.2) +
    geom_hline(yintercept = 0, linetype = "dashed", color = "gray") +
    labs(
      title = paste("Profit Curve:", class_label),
      subtitle = sprintf("TP=+%d | FP=%d | FN=%d", profit_tp, cost_fp, cost_fn),
      x = "% отобранных объектов", y = "Суммарная прибыль"
    ) +
    theme_minimal()
  
  return(list(profit = p_profit, data = profit_results))
}

compare_curves <- function(plot1, plot2, name1, name2, title, x_label, y_label) {
  data1 <- ggplot2::ggplot_build(plot1)$data[[1]]
  data2 <- ggplot2::ggplot_build(plot2)$data[[1]]
  data1$Model <- name1
  data2$Model <- name2
  combined <- bind_rows(data1, data2)
  combined$Model <- factor(combined$Model, levels = c(name1, name2))
  model_colors <- setNames(c("steelblue", "red"), c(name1, name2))
  
  ggplot(combined, aes(x = x, y = y, color = Model)) +
    geom_line(size = 1.2) +
    labs(title = title, subtitle = paste(name1, "vs", name2),
         x = x_label, y = y_label, color = "Модель") +
    scale_color_manual(values = model_colors) +
    theme_minimal() +
    theme(
      plot.title = element_text(hjust = 0.5, face = "bold"),
      legend.position = "top"
    )
}

# Хелперы для извлечения метрик из ggplot-объектов
extract_gain_at <- function(gain_plot, pct) {
  d <- ggplot_build(gain_plot)$data[[1]]
  d$y[which.min(abs(d$x - pct))]
}

extract_max_lift <- function(lift_plot) {
  d <- ggplot_build(lift_plot)$data[[1]]
  max(d$y, na.rm = TRUE)
}

extract_pct_at_gain <- function(gain_plot, target_gain) {
  d <- ggplot_build(gain_plot)$data[[1]]
  d$x[which.min(abs(d$y - target_gain))]
}
